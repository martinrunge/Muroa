cmake_minimum_required (VERSION 2.6)

set(CMAKE_MODULE_PATH "${CMAKE_CURRENT_SOURCE_DIR}/../cmake" ${CMAKE_MODULE_PATH})

include( MacroGDBIndex )

find_package ( Qt4 REQUIRED COMPONENTS QtCore QtGui )
find_package ( CppUnit REQUIRED )

set(Boost_USE_MULTITHREADED ON )
find_package( Boost 1.36.0 COMPONENTS filesystem REQUIRED )

add_definitions( ${QT4_DEFINITIONS} -D__STDC_CONSTANT_MACROS ${CPPUNIT_DEFINITIONS})

set( unittests_SRCS   CUtilsTest.cpp
                      CCollectionTest.cpp
                      CDiffTest.cpp
                      CRootItemDiffTest.cpp
                      CIteratorTest.cpp
                      CMuroaXmlRpcTest.cpp
                      CRpcDummy.cpp
                      LibMuroaTestApp.cpp
                      )
                    
set( unittests_H   
                      )
                    
 set( HEADERS         CUtilsTest.h
                      CCollectionTest.h
                      CDiffTest.h
                      CRootItemDiffTest.h
                      CIteratorTest.h
                      CMuroaXmlRpcTest.h
                      CRpcDummy.h
                      )                    
  

include ( ${QT_USE_FILE} )
qt4_wrap_cpp( MOC_SOURCES ${MOC_H} ${MOC_MEDPROC_H} )

include_directories(${CMAKE_CURRENT_SOURCE_DIR}/.. 
                    ${CMAKE_CURRENT_SOURCE_DIR}/../../libmcommons
                    ${Boost_INCLUDE_DIRS} 
                    ${libxdiff_INCLUDE_DIRS} 
                    ${CMAKE_CURRENT_BINARY_DIR}/..
                    ${CPPUNIT_INCLUDE_DIRS} ) 

link_directories(${BINARY_DIR}  ${QT_LIB_DIR} "${CMAKE_SOURCE_DIR}/../build" )

ADD_EXECUTABLE(libmuroatest ${unittests_SRCS} ${unittests_H} ${HEADERS} )

set_target_properties( libmuroatest PROPERTIES  COMPILE_FLAGS "-std=gnu++0x" ) 

TARGET_LINK_LIBRARIES(libmuroatest ${CPPUNIT_LIBRARIES} 
                                   muroa++
                                   mcommons
                                   ${Boost_LIBRARIES} 
                                   ${QT_LIBRARIES} 
                                   ${libxdiff_LIBRARIES}
                                   qxcppunitd 
                                   qxrunnerd
                                   )

ADD_GDB_INDEX( libmuroatest )
